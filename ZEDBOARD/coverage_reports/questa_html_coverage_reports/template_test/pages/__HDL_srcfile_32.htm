     66              channels           <= iFrameData.sobel;

     68              channels           <= iFrameData.embos;
     69          elsif(vChannelSelect = 8)then

     93          elsif(vChannelSelect = 20)then
     94              channels           <= iFrameData.tPattern;
     95          elsif(vChannelSelect = 21)then
     96              channels           <= iFrameData.cgainToCgain;
     97          elsif(vChannelSelect = 22)then
     98              channels           <= iFrameData.cgainToHsl;
     99          elsif(vChannelSelect = 23)then
     100              channels           <= iFrameData.cgainToHsv;

     115          elsif(vChannelSelect = 31)then

     131          elsif(vChannelSelect = 39)then
     132              channels           <= iFrameData.bluToCgaShp;
     133          elsif(vChannelSelect = 40)then

     136              channels           <= iFrameData.bluToCgaShpHsv;
     137          elsif(vChannelSelect = 42)then
     138              channels           <= iFrameData.bluToShpCga;
     139          elsif(vChannelSelect = 43)then
     140              channels           <= iFrameData.bluToShpCgaYcc;
