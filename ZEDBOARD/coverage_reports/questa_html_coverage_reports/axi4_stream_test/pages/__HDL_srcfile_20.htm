     2  library ieee;
     3  use ieee.std_logic_1164.all;
     4  use ieee.numeric_std.all;
     5  use work.constantspackage.all;
     6  use work.vpfRecords.all;
     7  use work.portspackage.all;
     8  entity frameTestPattern is
     9  generic (
     10      s_data_width	       : integer := 16);
     11  port (
     12      clk                   : in std_logic;
     13      iValid                : in std_logic;
     14      iCord                 : in coord;
     15      oRgb                  : out tpRgb);
     16  end frameTestPattern;
     17  architecture arch_imp of frameTestPattern is
     18      signal xCounter           : integer;
     19      signal yCounter           : integer;
     20      signal rowdist            : integer;
     21      signal nrowdist           : integer;
     22      signal coldist            : integer;
     23      signal ncoldist           : integer;
     24      signal irgbSum            : integer;
     25  begin
     26      xCounter    <= to_integer(unsigned(iCord.x));
     27      yCounter    <= to_integer(unsigned(iCord.y));
     28  tPattenP: Process (clk) begin
     29      if rising_edge(clk) then
     30          if iValid = '1'  then
     31              if xcounter > 960 then
     32                  rowdist <= xcounter - 960;
     33              else
     34                  rowdist <= 960 - xcounter;
     35              end if;
     36              if rowdist > 480 then
     37                  nrowdist <= rowdist - 480;
     38              else
     39                  nrowdist <= 480 - rowdist;
     40              end if;	
     41              if ycounter > 540 then
     42                  coldist <= ycounter -540;
     43              else
     44                  coldist <= 540 - ycounter;
     45              end if;
     46              if coldist > 270 then
     47                  ncoldist <= coldist - 270;
     48              else
     49                  ncoldist <= 270 - coldist;
     50              end if;
     51              irgbSum <= nrowdist + ncoldist;
     52          end if;
     53      end if;
     54  end process tPattenP;
     55  rgbSumP: Process (clk) begin
     56      if rising_edge(clk) then
     57          oRgb.valid    <= iValid;
     58          oRgb.red      <= std_logic_vector(to_unsigned(irgbSum,12));
     59          oRgb.green    <= std_logic_vector(to_unsigned(irgbSum,12));
     60          oRgb.blue     <= std_logic_vector(to_unsigned(irgbSum,12));
     61      end if;
     62  end process rgbSumP;
     63  end arch_imp;
