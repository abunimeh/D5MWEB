1  library ieee;
     2  use ieee.std_logic_1164.all;
     3  use ieee.numeric_std.all;
     4  use work.constantspackage.all;
     5  use work.vpfRecords.all;
     6  use work.portspackage.all;
     7  entity FrameMask is
     8  generic (
     9      eBlack         : boolean := false);
     10  port (
     11      clk            : in std_logic;
     12      reset          : in  std_logic;
     13      iEdgeValid     : in std_logic;
     14      i1Rgb          : in channel;
     15      i2Rgb          : in channel;
     16      oRgb           : out channel);
     17  end FrameMask;
     18  architecture behavioral of FrameMask is
     19      signal d1Rgb     : channel;
     20  begin
     21  SyncFrames32Inst: SyncFrames
     22  generic map(
     23      pixelDelay => 31) --LATENCY 32
     24  port map(
     25      clk        => clk,
     26      reset      => reset,
     27      iRgb       => i2Rgb,
     28      oRgb       => d1Rgb);
     29  EBLACK_ENABLED: if (eBlack = true) generate
     30      process (clk) begin
     31          if rising_edge(clk) then
     32              if (iEdgeValid = hi) then
     33                  oRgb.red   <= black;
     34                  oRgb.green <= black;
     35                  oRgb.blue  <= black;
     36              else
     37                  oRgb.red   <= d1Rgb.red;
     38                  oRgb.green <= d1Rgb.green;
     39                  oRgb.blue  <= d1Rgb.blue;
     40              end if;
     41                  oRgb.valid <= i1Rgb.valid;
     42          end if;
     43      end process;
     44  end generate EBLACK_ENABLED;
     45  EBLACK_DISABLED: if (eBlack = false) generate
     46      process (clk) begin
     47          if rising_edge(clk) then
     48              if (iEdgeValid = hi) then
     49                  oRgb.red   <= i1Rgb.red;
     50                  oRgb.green <= i1Rgb.green;
     51                  oRgb.blue  <= i1Rgb.blue;
     52              else
     53                  oRgb.red   <= d1Rgb.red;
     54                  oRgb.green <= d1Rgb.green;
     55                  oRgb.blue  <= d1Rgb.blue;
     56              end if;
     57                  oRgb.valid <= i1Rgb.valid;
     58          end if;
     59      end process;
     60  end generate EBLACK_DISABLED;
     61  end behavioral;
